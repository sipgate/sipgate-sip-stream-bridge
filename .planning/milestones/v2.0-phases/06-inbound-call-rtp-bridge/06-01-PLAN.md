Key sipgo v1.2.0 APIs (verified on pkg.go.dev):
```go
// DialogServerCache — manages UAS dialog state machine (INVITE → ACK → BYE)
dialogSrv := sipgo.NewDialogServerCache(agent.Client, contactHeader)

// ReadInvite MUST be called first in OnInvite — creates dialog, returns *DialogServerSession
dlg, err := dialogSrv.ReadInvite(req, tx)
defer dlg.Close() // always: removes from cache

// Sending responses via dialog (not tx.Respond — dlg tracks To-tag)
dlg.Respond(100, "Trying", nil)
dlg.RespondSDP(sdpAnswerBytes)    // sends 200 OK with SDP body; returns error if transport fails

// Terminating a call from our side
dlg.Bye(context.Background())     // sends BYE to caller; cancels dlg.Context()

// Routing ACK and BYE to the correct dialog (call-id + tag matching)
dialogSrv.ReadAck(req, tx)        // transitions to DialogStateConfirmed
dialogSrv.ReadBye(req, tx)        // sends 200 OK for BYE; cancels dlg.Context()

// Dialog context — cancelled when BYE arrives from either side
<-dlg.Context().Done()

// ContactHeader construction for NewDialogServerCache
contact := siplib.ContactHeader{
    Address: siplib.Uri{
        User: cfg.SIPUser,
        Host: cfg.SDPContactIP,
        Port: 5060,
    },
}

// siplib.Request header extraction
req.CallID().Value()          // string — SIP Call-ID header value
req.From().Address.String()   // string — From URI
req.To().Address.String()     // string — To URI
req.Body()                    // []byte — SDP offer body

// Response from raw tx (for OPTIONS — no dialog needed)
tx.Respond(siplib.NewResponseFromRequest(req, 200, "OK", nil))
```
